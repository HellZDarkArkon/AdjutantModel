// Manner of articulation — used by PhonemeSynth to select synthesis path
enum class ConsonantManner
{
    NOT_CONSONANT = -1,
    NASAL         = 0,
    PLOSIVE       = 1,
    SIBAFF        = 2,  // Sibilant affricate
    AFFRICATE     = 3,  // Non-sibilant affricate
    SIBFRIC       = 4,  // Sibilant fricative
    FRICATIVE     = 5,  // Non-sibilant fricative
    APPROXIMANT   = 6,
    FLAP          = 7,
    TRILL         = 8,
    LATAFF        = 9,  // Lateral affricate
    LATFRIC       = 10, // Lateral fricative
    LATAPP        = 11, // Lateral approximant
    LATFLAP       = 12, // Lateral flap
};

struct PhonemeData
{
    double f1 = 0.0, f2 = 0.0, f3 = 0.0, f4 = 0.0, f5 = 0.0;
    double bw1 = 0.0, bw2 = 0.0, bw3 = 0.0, bw4 = 0.0, bw5 = 0.0;
    // Consonant-specific parameters (0.0 default is safe for vowels)
    double noiseLevel    = 0.0; // 0.0=none, 1.0=full frication noise
    double noiseFreq     = 0.0; // Centre frequency of frication noise band (Hz)
    double noiseBw       = 0.0; // Bandwidth of frication noise band (Hz)
    double burstDuration = 0.0; // Plosive burst length (ms); 0 = non-plosive
    double voicingRatio  = 0.0; // 0.0=fully voiceless, 1.0=fully voiced
};

class Phoneme
{
public:
    static void LoadPhonemeData(Framework& fw, std::string& filePath = *(new std::string("")));

    Phoneme(PhonemeType t) : type(t), f1(0.0), f2(0.0), f3(0.0), f4(0.0), f5(0.0),
        bw1(0.0), bw2(0.0), bw3(0.0), bw4(0.0), bw5(0.0),
        noiseLevel(0.0), noiseFreq(0.0), noiseBw(0.0), burstDuration(0.0), voicingRatio(0.0)
    {
        auto it = s_phonemeData.find(t);
        if (it != s_phonemeData.end())
        {
            const PhonemeData& d = it->second;
            f1 = d.f1; f2 = d.f2; f3 = d.f3; f4 = d.f4; f5 = d.f5;
            bw1 = d.bw1; bw2 = d.bw2; bw3 = d.bw3; bw4 = d.bw4; bw5 = d.bw5;
            noiseLevel    = d.noiseLevel;
            noiseFreq     = d.noiseFreq;
            noiseBw       = d.noiseBw;
            burstDuration = d.burstDuration;
            voicingRatio  = d.voicingRatio;
        }
    }

    PhonemeType GetType() const { return type; }
    double GetFormant1() const { return f1; }
    double GetFormant2() const { return f2; }
    double GetFormant3() const { return f3; }
    double GetFormant4() const { return f4; }
    double GetFormant5() const { return f5; }
    double GetBandwidth1() const { return bw1; }
    double GetBandwidth2() const { return bw2; }
    double GetBandwidth3() const { return bw3; }
    double GetBandwidth4() const { return bw4; }
    double GetBandwidth5() const { return bw5; }
    double GetNoiseLevel()    const { return noiseLevel; }
    double GetNoiseFreq()     const { return noiseFreq; }
    double GetNoiseBw()       const { return noiseBw; }
    double GetBurstDuration() const { return burstDuration; }
    double GetVoicingRatio()  const { return voicingRatio; }

    bool IsConsonant() const { return type >= PhonemeType::NASAL_BILABIAL; }

    ConsonantManner GetManner() const
    {
        if (!IsConsonant()) return ConsonantManner::NOT_CONSONANT;
        if (type <= PhonemeType::NASAL_UVULAR)                  return ConsonantManner::NASAL;
        if (type <= PhonemeType::PLOSIVE_GLOTTAL)               return ConsonantManner::PLOSIVE;
        if (type <= PhonemeType::SIBAFF_PALATAL_VOICED)         return ConsonantManner::SIBAFF;
        if (type <= PhonemeType::AFFRICATE_DENTAL_VOICED)       return ConsonantManner::AFFRICATE;
        if (type <= PhonemeType::SIBFRIC_PALATAL_VOICED)        return ConsonantManner::SIBFRIC;
        if (type <= PhonemeType::FRICATIVE_GLOTTAL_VOICED)      return ConsonantManner::FRICATIVE;
        if (type <= PhonemeType::APPROX_LABIOVELAR)             return ConsonantManner::APPROXIMANT;
        if (type <= PhonemeType::FLAP_RETROFLEX)                return ConsonantManner::FLAP;
        if (type <= PhonemeType::TRILL_UVULAR)                  return ConsonantManner::TRILL;
        if (type <= PhonemeType::LATAFF_ALVEOLAR_VOICED)        return ConsonantManner::LATAFF;
        if (type <= PhonemeType::LATFRIC_ALVEOLAR_VOICED)       return ConsonantManner::LATFRIC;
        if (type <= PhonemeType::LATAPP_VELAR)                  return ConsonantManner::LATAPP;
        return ConsonantManner::LATFLAP;
    }
