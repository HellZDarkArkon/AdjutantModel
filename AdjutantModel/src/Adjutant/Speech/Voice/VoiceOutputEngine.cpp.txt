#include "VoiceOutputEngine.h"
#include <iostream>

VoiceOutputEngine::VoiceOutputEngine() : pVoice(nullptr), mSampleRate(0), comInitialized(false)
{
	HRESULT hr = CoInitialize(NULL); // Initialize COM library
	if (FAILED(hr))
	{
		std::cout << "[VOICE] CoInitialize failed. hr = 0x" << std::hex << hr << std::endl;
		pVoice = nullptr; // Ensure pVoice is null if COM initialization fails
		return;
	}
	comInitialized = true;

	hr = CoCreateInstance(
		CLSID_SpVoice,
		NULL,
		CLSCTX_ALL,
		IID_ISpVoice,
		(void**)&pVoice
	);

	if (FAILED(hr))
	{
		std::cout << "[VOICE] CocreateInstance(CLSID_SpVoice) FAILED. hr = 0x" << std::hex << hr << std::endl;
		pVoice = nullptr; // Ensure pVoice is null if voice creation fails
	}
}

VoiceOutputEngine::~VoiceOutputEngine()
{
	Clean(); // Clean up resources when the voice output engine is destroyed
}

void VoiceOutputEngine::Clean()
{
	if (pVoice)
	{
		pVoice->Release(); // Release the SAPI voice interface
		pVoice = nullptr; // Set pVoice to null after releasing
	}
	if (comInitialized)
	{
		CoUninitialize(); // Uninitialize COM library only if this instance owns the call
		comInitialized = false;
	}
}

double VoiceOutputEngine::SpeakTimed(const std::wstring& text)
{
	if (!pVoice)
		return 0.0; // Return 0 if the voice engine is not initialized
	LARGE_INTEGER freq, start, end;
	QueryPerformanceFrequency(&freq); // Get the frequency of the high-resolution performance counter
	QueryPerformanceCounter(&start); // Get the starting timestamp

	// Speak the text synchronously to measure the time taken for speech output
	HRESULT hr = pVoice->Speak(text.c_str(), SPF_DEFAULT, NULL); // Speak the text synchronously

	QueryPerformanceCounter(&end); // Get the ending timestamp

	if (FAILED(hr))
	{
		std::cout << "[VOICE] Speech failed. hr = 0x" << std::hex << hr << std::endl;
		return 0.0; // Return 0 if speech failed
	}

	double seconds = double(end.QuadPart - start.QuadPart) / double(freq.QuadPart); // Calculate the time taken for speech output in seconds
	return seconds; // Return the time taken for speech output
}

void VoiceOutputEngine::PlayPCM(const short* samples, size_t count)
{
	
	IMMDeviceEnumerator* enumerator = nullptr;
	IMMDevice* device = nullptr;

	HRESULT hr = CoCreateInstance(__uuidof(MMDeviceEnumerator), NULL,
		CLSCTX_ALL, __uuidof(IMMDeviceEnumerator), (void**)&enumerator); // Create an instance of the MMDeviceEnumerator to access audio devices

	if (FAILED(hr) || !enumerator)
	{
		std::cout << "[VOICE] CoCreateInstance for MMDeviceEnumerator failed. hr = 0x" << std::hex << hr << std::endl;
		return;
	}
	enumerator->GetDefaultAudioEndpoint(eRender, eConsole, &device); // Get the default audio endpoint for rendering (playback)

	if(!device) {
		std::cout << "[VOICE] GetDefaultAudioEndpoint failed." << std::endl;
		enumerator->Release(); // Release the device enumerator
		return;
	}

	IAudioClient* audioClient = nullptr;
	device->Activate(__uuidof(IAudioClient), CLSCTX_ALL,
		NULL, (void**)&audioClient); // Activate the audio client for the selected audio device

	if(!audioClient)
	{
		std::cout << "[VOICE] Activate for IAudioClient failed." << std::endl;
		device->Release(); // Release the audio device
		enumerator->Release(); // Release the device enumerator
		return;
	}

	WAVEFORMATEX* wfx = nullptr; // Structure to hold the audio format information
	audioClient->GetMixFormat(&wfx); // Get the mix format of the audio device to ensure compatibility with the audio data we will be playing
	if(!(GetSampleRate() == wfx->nSamplesPerSec))
		SetSampleRate(wfx->nSamplesPerSec); // Update the sample rate in the voice output engine to match the device's mix format sample rate
	std::cout << "[VOICE] Device Mix Format: " << wfx->nChannels << " channels, " << wfx->nSamplesPerSec << " Hz, " << wfx->wBitsPerSample << " bits per sample." << std::endl;

	hr = audioClient->Initialize(
	AUDCLNT_SHAREMODE_SHARED,
		0, 
		10000000, // Buffer duration in 100-nanosecond units (10 seconds in this case)
		0,
		wfx,
		NULL); // Initialize the audio client in shared mode with the specified format

	if (FAILED(hr))
	{
		std::cout << "[VOICE] Audio client initialization failed. hr = 0x" << std::hex << hr << std::endl;
		CoTaskMemFree(wfx); // Free the memory allocated for the wave format structure
		audioClient->Release(); // Release the audio client
		device->Release(); // Release the audio device
		enumerator->Release(); // Release the device enumerator
		return;
	}

	IAudioRenderClient* renderClient = nullptr;
	audioClient->GetService(__uuidof(IAudioRenderClient), (void**)&renderClient); // Get the audio render client to write audio data to the buffer

	UINT32 bufferFrameCount;
	audioClient->GetBufferSize(&bufferFrameCount); // Get the size of the audio buffer in frames

	BYTE* data = nullptr;
	UINT32 framestoWrite = (UINT32)count; // Calculate the number of frames to write based on the count of samples (assuming 16-bit stereo, so 4 bytes per frame)
	if(framestoWrite > bufferFrameCount)
		framestoWrite = bufferFrameCount; // Ensure we don't write more frames than the buffer can hold
	
	hr = renderClient->GetBuffer(framestoWrite, &data); // Get a pointer to the audio buffer where we will write the audio data
	if(FAILED(hr) || data == nullptr)
	{
		std::cout << "[VOICE] GetBuffer failed. hr = 0x" << std::hex << hr << std::endl;
		audioClient->Stop(); // Stop the audio stream
		renderClient->Release(); // Release the audio render client
		CoTaskMemFree(wfx); // Free the memory allocated for the wave format structure
		audioClient->Release(); // Release the audio client
		device->Release(); // Release the audio device
		enumerator->Release(); // Release the device enumerator
		return;
	}

	float* floatData = reinterpret_cast<float*>(data); // Cast the byte pointer to a float pointer for writing audio data in the correct format
	for (UINT32 i = 0; i < framestoWrite; i++)
	{
		float sample = samples[i] / 32768.0f; // Convert the 16-bit PCM samples to floating-point format in the range [-1.0, 1.0]
		for (UINT32 ch = 0; ch < wfx->nChannels; ch++)
		{
			floatData[i * wfx->nChannels + ch] = sample; // Write the sample to each channel
		}
	}

	renderClient->ReleaseBuffer(framestoWrite, 0); // Release the buffer after writing the audio data

	audioClient->Start(); // Start the audio stream to begin playback of the audio data we just wrote to the buffer

	Sleep((DWORD)((framestoWrite / (double)wfx->nSamplesPerSec) * 1000.0)); // Sleep for the duration of the audio being played to ensure it finishes before we clean up resources

	audioClient->Stop(); // Stop the audio stream
	renderClient->Release(); // Release the audio render client
	CoTaskMemFree(wfx); // Free the memory allocated for the wave format structure
	audioClient->Release(); // Release the audio client
	device->Release(); // Release the audio device
	enumerator->Release(); // Release the device enumerator
}

void VoiceOutputEngine::ListVoices()
{
	IEnumSpObjectTokens* cpEnum = nullptr;
	ULONG count = 0;

	SpEnumTokens(SPCAT_VOICES, NULL, NULL, &cpEnum); // Enumerate available voices in the SAPI voice engine
	cpEnum->GetCount(&count); // Get the count of available voices

	for (ULONG i = 0; i < count; i++)
	{
		ISpObjectToken* cpVoiceToken = nullptr;
		cpEnum->Next(1, &cpVoiceToken, NULL); // Get the next voice token

		LPWSTR description = nullptr;
		SpGetDescription(cpVoiceToken, &description);

		wprintf(L"Voice %u: %s\n", i, description); // Print the description of the voice

		CoTaskMemFree(description); // Free the memory allocated for the description
		cpVoiceToken->Release(); // Release the voice token
	}
	cpEnum->Release(); // Release the enumerator
}

VoiceOutputEngine::VoiceOutputEngine(VoiceOutputEngine&& other) noexcept
	: pVoice(other.pVoice), mSampleRate(other.mSampleRate), comInitialized(other.comInitialized)
{
	other.pVoice = nullptr;
	other.comInitialized = false; // Source no longer owns the CoInitialize call
}

VoiceOutputEngine& VoiceOutputEngine::operator=(VoiceOutputEngine&& other) noexcept
{
	if (this != &other)
	{
		Clean();
		pVoice         = other.pVoice;
		mSampleRate    = other.mSampleRate;
		comInitialized = other.comInitialized;
		other.pVoice         = nullptr;
		other.comInitialized = false; // Source no longer owns the CoInitialize call
	}
	return *this;
}
