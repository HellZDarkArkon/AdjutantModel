class FileLoader
{
public:
	FileLoader(const std::string& fPath="")
	{
		mFilePath = fPath; // Store the file path for later use
	}

	//Load a file and return its contents as a string
	std::string LoadFile(const std::string& filePath)
	{
		std::ifstream file(filePath); // Open the file for reading
		if (!file.is_open())
		{
			std::cerr << "Failed to open file: " << filePath << std::endl; // Print an error message if the file cannot be opened
			return ""; // Return an empty string if the file cannot be opened
		}
		std::stringstream buffer; // Create a stringstream to hold the file contents
		buffer << file.rdbuf(); // Read the entire file into the stringstream
		return buffer.str(); // Return the contents of the stringstream as a string
	}

	//Load a file and target a specific section based on #<tag> markers on new lines, taking the data on each line in the section with each value between - characters as a separate variable number, returning a vector of vectors of strings for each section
	std::vector<std::vector<std::string>> LoadSection(const std::string& filePath, const std::string& sectionTag)
	{
		std::ifstream file(filePath); // Open the file for reading
		if (!file.is_open())
		{
			std::cerr << "Failed to open file: " << filePath << std::endl; // Print an error message if the file cannot be opened
			return {}; // Return an empty vector if the file cannot be opened
		}
		std::vector<std::vector<std::string>> sectionData; // Vector to hold the data for the specified section
		std::string line; // Variable to hold each line read from the file
		bool inSection = false; // Flag to indicate whether we are currently within the specified section
		while (std::getline(file, line)) // Read the file line by line
		{
			if (line == "#" + sectionTag) // Check if the line is the start of the specified section
			{
				inSection = true; // Set the flag to indicate we are now in the section
				continue; // Skip to the next line
			}
			if (inSection)
			{
				if (line.empty() || line[0] == '#') // Check if we have reached the end of the section (empty line or new section marker)
				{
					break; // Exit the loop if we have reached the end of the section
				}
				std::vector<std::string> values; // Vector to hold the values for this line
				std::stringstream ss(line); // Create a stringstream to parse the line
				std::string value; // Variable to hold each value parsed from the line
				while (std::getline(ss, value, '-')) // Split the line into values based on '-' delimiter
				{
					values.push_back(value); // Add each value to the vector for this line
				}
				sectionData.push_back(values); // Add the vector of values for this line to the section data vector
			}
		}
		return sectionData; // Return the data for the specified section as a vector of vectors of strings
	}

	std::vector<std::string> loadDatabaseSection(const std::string& fPath, const std::string& sectionTag)
	{
		std::ifstream file(fPath);
		if (!file.is_open())
		{
			std::cerr << "Failed to open file: " << fPath << std::endl;
			return {};
		}
		std::vector<std::string> sectionData;
		std::string line;
		bool inSection = false;
		while (std::getline(file, line))
		{
			if (line == "#" + sectionTag)
			{
				inSection = true;
				continue;
			}
			if (inSection)
			{
				if (line.empty()) continue;
				if (line[0] == '#')
					break;

				std::string encoded;

				for (size_t i = 0; i < line.size(); i++)
				{
					size_t pipePos = line.find('|');
					if (pipePos != std::string::npos)
					{
						std::string left = line.substr(0, pipePos);
						std::stringstream ss(left);
						std::string token;
						while (ss >> token)
							encoded += "[CONST:" + token + "]";
						i = pipePos + 1;
					}
					std::string arg;
					std::vector<std::string> argTokens;
					int parenDepth = 0;
					while (i < line.size())
					{
						if (line[i] == '(') parenDepth++;
						else if (line[i] == ')' && parenDepth > 0) parenDepth--;

						if (parenDepth == 0 && line[i] == ' ' && i + 2 < line.size() && line[i + 1] == '-' && line[i + 2] == ' ')
						{
							if (!arg.empty()) { argTokens.push_back(arg); arg.clear(); }
							i += 3; // skip ' - '
							continue;
						}
						arg += line[i++];
					}
					if (!arg.empty()) argTokens.push_back(arg);

					for (auto& a : argTokens)
						encoded += "[ARG:" + ParseArgExpression(a) + "]";
				}
				sectionData.push_back(encoded); // collect the encoded line
			}
		}
		return sectionData; // return all encoded lines
	}

	std::string LoadShaderSection(const std::string& fPath, const std::string& sectionTag)
	{
		std::ifstream file(fPath); // Open the file for reading
		if(!file.is_open())
		{
			std::cerr << "Failed to open file: " << fPath << std::endl; // Print an error message if the file cannot be opened
			return ""; // Return an empty string if the file cannot be opened
		}

		std::string res;
		std::string line;
		bool inSection = false; // Flag to indicate whether we are currently within the specified section

		while (std::getline(file, line))
		{
			if (line == "#" + sectionTag) // Check if the line is the start of the specified section
			{
				inSection = true; // Set the flag to indicate we are now in the section
				continue; // Skip to the next line
			}

			if (inSection)
			{
				if (line.size() > 0 && line[0] == '#' && line.find(' ') == std::string::npos)
					break;

				res += line + "\n"; // Add the line to the result string with a newline character
			}
		}
		return res;
	}

	void SetFilePath(const std::string& filePath)
	{
		mFilePath = filePath; // Update the file path for later use
	}

	std::string GetFilePath() const
	{
		return mFilePath; // Return the current file path
	}

	//Utility function to get individual data sections from a output file section of LoadSection() by index between - characters, returning a vector of strings for the specified index across that line of the section.
	std::vector<std::string> GetSectionDataByIndex(const std::vector<std::vector<std::string>>& sectionData, size_t index)
	{
		std::vector<std::string> result; // Vector to hold the values for the specified index
		for (const auto& lineValues : sectionData) // Iterate through each line of the section data
		{
			if (index < lineValues.size()) // Check if the specified index is within the bounds of the current line's values
			{
				result.push_back(lineValues[index]); // Add the value at the specified index to the result vector
			}
			else
			{
				result.push_back(""); // If the index is out of bounds for this line, add an empty string to maintain alignment
			}
		}
		return result; // Return the vector of values for the specified index across all lines of the section
	}
	// Utility function to convert a vector of strings from a specific index of a section data to a vector of doubles, handling any conversion errors gracefully by returning 0.0 for invalid entries. (should turn "1.23" into 1.23, and "abc" into 0.0 with an error message)
	double ConvertSectionDataStringToDouble(const std::vector<std::string>& data, size_t index)
	{
		if (index < data.size()) // Check if the specified index is within the bounds of the data vector
		{
			try
			{
				return std::stod(data[index]); // Convert the string at the specified index to a double and return it
			}
			catch (const std::invalid_argument& e)
			{
				std::cerr << "Invalid argument for conversion to double: " << data[index] << " at index " << index << std::endl; // Print an error message if the conversion fails due to an invalid argument
				return 0.0; // Return 0.0 if the conversion fails
			}
			catch (const std::out_of_range& e)
			{
				std::cerr << "Out of range error for conversion to double: " << data[index] << " at index " << index << std::endl; // Print an error message if the conversion fails due to an out-of-range value
				return 0.0; // Return 0.0 if the conversion fails
			}
		}
		else
		{
			std::cerr << "Index out of bounds for data vector: " << index << std::endl; // Print an error message if the specified index is out of bounds for the data vector
			return 0.0; // Return 0.0 if the index is out of bounds
		}
	}

	static void RemoveSection(std::string& src, const std::string& tag)
	{
		const std::string marker = "#" + tag; // Construct the section marker string based on the specified tag

		size_t start = src.find(marker);
		if (start == std::string::npos) return; // If the section marker is not found, return without modifying the source string
		
		size_t next = src.find("#", start + marker.length()); // Find the next section marker after the current one

		if(next == std::string::npos) // If there is no next section marker, remove everything from the current marker to the end of the string
			src.erase(start);
		else // Otherwise, remove the section from the current marker to the next marker
			src.erase(start, next - start);
	}

	// Load a single person's key-value fields from a targetpeople-style database section.
	// Parses <VAR % VALUE> bracket tokens directly — does NOT split on '-'.
	std::unordered_map<std::string, std::string> loadPersonData(
		const std::string& fPath,
		const std::string& sectionTag,
		const std::string& targetConstant)
	{
		std::unordered_map<std::string, std::string> result;
		std::ifstream file(fPath);
		if (!file.is_open())
		{
			std::cerr << "Failed to open file: " << fPath << std::endl;
			return result;
		}
		auto trim = [](std::string s) -> std::string {
			size_t first = s.find_first_not_of(" \t");
			size_t last  = s.find_last_not_of(" \t");
			return (first == std::string::npos) ? "" : s.substr(first, last - first + 1);
		};
		std::string line;
		bool inSection = false;
		while (std::getline(file, line))
		{
			if (line == "#" + sectionTag)
			{
				inSection = true;
				continue;
			}
			if (!inSection) continue;
			if (line.empty()) continue;
			if (line[0] == '#') break;

			size_t pipePos = line.find('|');
			if (pipePos == std::string::npos) continue;

			// Check left-side constants for targetConstant
			std::string left = line.substr(0, pipePos);
			bool found = false;
			std::stringstream ss(left);
			std::string tok;
			while (ss >> tok)
				if (tok == targetConstant) { found = true; break; }
			if (!found) continue;

			// Scan <VAR % VALUE> bracket tokens from the right side
			std::string right = line.substr(pipePos + 1);
			size_t i = 0;
			while (i < right.size())
			{
				size_t open = right.find('<', i);
				if (open == std::string::npos) break;
				size_t close = right.find('>', open);
				if (close == std::string::npos) break;
				std::string token = right.substr(open + 1, close - open - 1);
				size_t pct = token.find('%');
				if (pct != std::string::npos)
				{
					std::string key = trim(token.substr(0, pct));
					std::string val = trim(token.substr(pct + 1));
					if (val.size() >= 2 && val.front() == '"' && val.back() == '"')
						val = val.substr(1, val.size() - 2);
					result[key] = val;
				}
				i = close + 1;
			}
			break;
		}
		return result;
	}

private:
	std::string mFilePath; // Member variable to store the file path for later use

	// Recursively encodes one argument expression from the right-hand side of a database line.
	// Handles <$VAR>(case0 || case1 || ...) with unlimited cases and unlimited nesting depth:
	//   depth counter gates || splits so inner parens never prematurely close the outer case;
	//   each branch is fed back through ParseArgExpression before being stored as [CASE:N=...].
	// Handles <VAR> (ADJUTANT_TAG or PRINT ENUM) and plain literal characters unchanged.
	std::string ParseArgExpression(const std::string& a)
	{
		std::string parsedArg;
		size_t j = 0;
		while (j < a.size())
		{
			if (a[j] == '<' && j + 1 < a.size() && a[j + 1] == '$')
				{
					j += 2; // skip '<$'
					std::string var;
					while (j < a.size() && a[j] != '>') var += a[j++];
					if (j < a.size()) j++; // skip '>'

					// <$VAR % VALUE> — declarative set side-effect, produces no display text
					size_t pct = var.find(" % ");
					if (pct != std::string::npos)
					{
						std::string varName = var.substr(0, pct);
						std::string value   = var.substr(pct + 3);
						parsedArg += "[SET:" + varName + "=" + value + "]";
						continue;
					}

					parsedArg += "[SWITCH:" + var + "]";
				while (j < a.size() && a[j] != '(') j++;
				if (j < a.size() && a[j] == '(')
				{
					j++; // skip opening '('
					std::string branch;
					int branchIndex = 0;
					int depth = 0;
					while (j < a.size() && !(a[j] == ')' && depth == 0))
					{
						if (a[j] == '(') { depth++; branch += a[j++]; continue; }
						if (a[j] == ')') { depth--; branch += a[j++]; continue; }
						if (a[j] == '|' && j + 1 < a.size() && a[j + 1] == '|' && depth == 0)
						{
							size_t s = branch.find_first_not_of(" \t");
							size_t e = branch.find_last_not_of(" \t");
							std::string trimmed = (s == std::string::npos) ? "" : branch.substr(s, e - s + 1);
							parsedArg += "[CASE:" + std::to_string(branchIndex++) + "=" + ParseArgExpression(trimmed) + "]";
							branch.clear();
							j += 2;
							continue;
						}
						branch += a[j++];
					}
					{
						size_t s = branch.find_first_not_of(" \t");
						size_t e = branch.find_last_not_of(" \t");
						std::string trimmed = (s == std::string::npos) ? "" : branch.substr(s, e - s + 1);
						if (!trimmed.empty())
							parsedArg += "[CASE:" + std::to_string(branchIndex) + "=" + ParseArgExpression(trimmed) + "]";
					}
					if (j < a.size() && a[j] == ')') j++; // skip closing ')'
				}
				continue;
			}
			if (a[j] == '<')
			{
				j++;
				std::string var;
				while (j < a.size() && a[j] != '>') var += a[j++];
				if (j < a.size()) j++; // skip '>'
				if (var == "ADJUTANT_TAG")
					parsedArg += "[ADJUTANT] >"; // compile-time constant — baked in at load time
				else
					parsedArg += "[PRINT ENUM:" + var + "]";
				continue;
			}
			parsedArg += a[j++];
		}
		return parsedArg;
	}

		std::string name;
	};

	};
